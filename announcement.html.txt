  <script src="js/api.js"></script>
  <script>
    const urlParams = new URLSearchParams(window.location.search);
    const userType = urlParams.get('userType') || localStorage.getItem('userType') || 'teacher';
    
    async function loadAnnouncements() {
      const response = await api.getAnnouncements();
      const container = document.getElementById('announcementList');
      if (response.success && response.announcements) {
        if (response.announcements.length === 0) { container.innerHTML = '<p>暫無公告</p>'; return; }
        container.innerHTML = response.announcements.map(a => `
          <div class="announcement-card">
            <div class="announcement-title">${a.title}</div>
            <div class="announcement-meta">📅 ${a.publishDate?.split('T')[0] || ''} | 類型：${a.type || '一般公告'}</div>
            <div>${a.content || ''}</div>
          </div>
        `).join('');
      } else { container.innerHTML = '<p>載入失敗</p>'; }
    }
    
    async function submitAnnouncement() {
      const data = { title: document.getElementById('annTitle').value, content: document.getElementById('annContent').value, type: document.getElementById('annType').value };
      if (!data.title || !data.content) { showMessage('請填寫標題和內容', 'error'); return; }
      const response = await api.saveAnnouncement(data);
      if (response.success) { showMessage('公告發布成功！', 'success'); document.getElementById('annTitle').value = ''; document.getElementById('annContent').value = ''; switchTab('list'); loadAnnouncements(); }
      else { showMessage('發布失敗：' + response.message, 'error'); }
    }
    
    function switchTab(tab) {
      document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
      document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
      if (tab === 'list') { document.querySelector('.tab:first-child').classList.add('active'); document.getElementById('list-tab').classList.add('active'); loadAnnouncements(); }
      else { document.querySelector('.tab:last-child').classList.add('active'); document.getElementById('add-tab').classList.add('active'); }
    }
    
    function showMessage(msg, type) { const msgDiv = document.getElementById('message'); msgDiv.textContent = msg; msgDiv.className = 'message ' + type; setTimeout(() => msgDiv.style.display = 'none', 3000); }
    function goBack() { window.location.href = `dashboard.html?userType=${userType}`; }
    loadAnnouncements();
  </script>
