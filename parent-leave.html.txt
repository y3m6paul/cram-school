  <script>
    async function submitLeave() {
      const data = {
        studentName: document.getElementById('studentName').value,
        leaveDate: document.getElementById('leaveDate').value,
        reason: document.getElementById('reason').value,
        applicant: document.getElementById('parentName').value,
        parentEmail: document.getElementById('parentEmail').value
      };
      if (!data.studentName || !data.leaveDate || !data.reason || !data.applicant || !data.parentEmail) {
        showMessage('請填寫所有欄位', 'error'); return;
      }
      const response = await api.applyLeave(data);
      if (response.success) { showMessage('請假申請已送出！', 'success'); setTimeout(() => goBack(), 2000); }
      else { showMessage('申請失敗：' + response.message, 'error'); }
    }
    function showMessage(msg, type) { const msgDiv = document.getElementById('message'); msgDiv.textContent = msg; msgDiv.className = 'message ' + type; setTimeout(() => msgDiv.style.display = 'none', 3000); }
    function goBack() { window.location.href = 'parent-dashboard.html'; }
    document.getElementById('leaveDate').value = new Date().toISOString().split('T')[0];
  </script>
