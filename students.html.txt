<script src="js/api.js"></script>
<script>
  let students = [];
  let selectedStudent = '';
  let homeworks = [];
  
  async function loadStudents() {
    const res = await api.getStudents();
    if (res.success) {
      students = res.students;
      const select = document.getElementById('studentSelect');
      students.forEach(s => {
        const option = document.createElement('option');
        option.value = s.englishName;
        option.textContent = `${s.englishName} (${s.chineseName}) - ${s.className}`;
        select.appendChild(option);
      });
    }
  }
  
  async function loadHomework() {
    if (!selectedStudent) return;
    const res = await api.getHomework();
    if (res.success) {
      homeworks = res.homeworks;
      const container = document.getElementById('homeworkList');
      
      let html = '';
      for (let hw of homeworks) {
        const check = await api.checkSubmission(hw.id, selectedStudent);
        html += `
          <div class="homework-item">
            <div class="homework-title">📌 ${hw.title}</div>
            <div class="homework-date">📅 發布：${new Date(hw.date).toLocaleString()}</div>
            <div>${hw.description}</div>
            ${hw.imageUrl ? `<img src="${hw.imageUrl}" style="max-width:150px; margin:10px 0;">` : ''}
            <div style="margin-top: 10px;">
              ${check.submitted ? 
                `<div class="status-submitted">✅ 已繳交！上傳時間：${new Date(check.uploadDate).toLocaleString()}</div>
                 <div class="submitted-file"><a href="${check.fileUrl}" target="_blank">📁 查看已上傳檔案</a></div>` :
                `<div class="status-not">⏳ 尚未繳交</div>
                 <input type="file" id="file_${hw.id}" accept=".mp3,.wav,.m4a,.jpg,.jpeg,.png">
                 <button onclick="uploadFile('${hw.id}')">📤 上傳作業</button>
                 <div id="status_${hw.id}" style="margin-top:5px;"></div>`
              }
            </div>
          </div>
        `;
      }
      container.innerHTML = html;
    }
  }
  
  async function uploadFile(homeworkId) {
    const fileInput = document.getElementById(`file_${homeworkId}`);
    const file = fileInput.files[0];
    if (!file) { alert('請選擇檔案'); return; }
    
    const isAudio = file.name.match(/\.(mp3|wav|m4a)$/i);
    const isImage = file.name.match(/\.(jpg|jpeg|png)$/i);
    if (!isAudio && !isImage) { alert('請上傳 MP3 或圖片檔案'); return; }
    
    const statusDiv = document.getElementById(`status_${homeworkId}`);
    statusDiv.innerHTML = '⏳ 上傳中...';
    
    const reader = new FileReader();
    reader.onload = async function(e) {
      const base64 = e.target.result.split(',')[1];
      const res = await api.uploadHomework(base64, file.name, homeworkId, selectedStudent);
      if (res.success) {
        statusDiv.innerHTML = '✅ 上傳成功！';
        setTimeout(() => loadHomework(), 2000);
      } else {
        statusDiv.innerHTML = '❌ 上傳失敗：' + res.message;
      }
    };
    reader.readAsDataURL(file);
  }
  
  document.getElementById('studentSelect').onchange = function() {
    selectedStudent = this.value;
    if (selectedStudent) loadHomework();
    else document.getElementById('homeworkList').innerHTML = '請先選擇姓名';
  };
  
  function logout() { localStorage.clear(); window.location.href = 'index.html'; }
  loadStudents();
</script>
