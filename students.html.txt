  <script>
    const urlParams = new URLSearchParams(window.location.search);
    const userType = urlParams.get('userType') || localStorage.getItem('userType') || 'teacher';
    
    async function loadClasses() {
      const response = await api.get({ action: 'getClassList' });
      if (response.success && response.classes) {
        const select = document.getElementById('classFilter');
        response.classes.forEach(c => { const opt = document.createElement('option'); opt.value = c; opt.textContent = c; select.appendChild(opt); });
      }
    }
    
    async function loadStudents() {
      const className = document.getElementById('classFilter').value;
      const response = await api.getStudents(className);
      if (response.success && response.students) {
        const search = document.getElementById('searchInput').value.toLowerCase();
        let filtered = response.students;
        if (search) filtered = filtered.filter(s => s.name?.toLowerCase().includes(search) || s.id?.toLowerCase().includes(search));
        document.getElementById('studentList').innerHTML = filtered.map(s => `<tr><td>${s.id}</td><td>${s.name}</td><td>${s.class || ''}</td><td>${s.motherPhone || ''}</td><td>${s.fatherPhone || ''}</td></tr>`).join('');
        if (filtered.length === 0) document.getElementById('studentList').innerHTML = '<tr><td colspan="5">無學生資料</td></tr>';
      } else { showMessage('載入失敗', 'error'); }
    }
    
    function showMessage(msg, type) { const msgDiv = document.getElementById('message'); msgDiv.textContent = msg; msgDiv.className = 'message ' + type; setTimeout(() => msgDiv.style.display = 'none', 3000); }
    function goBack() { window.location.href = `dashboard.html?userType=${userType}`; }
    loadClasses(); loadStudents();
  </script>
