  <script>
    const urlParams = new URLSearchParams(window.location.search);
    const userType = urlParams.get('userType') || localStorage.getItem('userType') || 'teacher';
    
    async function loadTeachers() {
      const response = await api.get({ action: 'getTeachers' });
      if (response.success && response.teachers) {
        const search = document.getElementById('searchInput').value.toLowerCase();
        let filtered = response.teachers;
        if (search) filtered = filtered.filter(t => t.name?.toLowerCase().includes(search) || t.subject?.toLowerCase().includes(search));
        document.getElementById('teacherList').innerHTML = filtered.map(t => `<tr><td>${t.id}</td><td>${t.name}</td><td>${t.subject}</td><td>${t.phone}</td><td>${t.email}</td></tr>`).join('');
        if (filtered.length === 0) document.getElementById('teacherList').innerHTML = '<tr><td colspan="5">無老師資料</td></tr>';
      } else { showMessage('載入失敗', 'error'); }
    }
    
    function showMessage(msg, type) { const msgDiv = document.getElementById('message'); msgDiv.textContent = msg; msgDiv.className = 'message ' + type; setTimeout(() => msgDiv.style.display = 'none', 3000); }
    function goBack() { window.location.href = `dashboard.html?userType=${userType}`; }
    loadTeachers();
  </script>
