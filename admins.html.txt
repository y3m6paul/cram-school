<script src="js/api.js"></script>
<script>
  let currentImageBase64 = '';
  let currentImageName = '';
  
  async function loadHomework() {
    const res = await api.getHomework();
    if (res.success && res.homeworks.length) {
      document.getElementById('homeworkList').innerHTML = res.homeworks.map(h => `
        <div class="homework-item">
          <div class="homework-title">📌 ${h.title}</div>
          <div class="homework-date">📅 發布：${new Date(h.date).toLocaleString()}</div>
          <div>${h.description}</div>
          ${h.imageUrl ? `<img src="${h.imageUrl}" class="homework-img">` : ''}
          <button class="btn-danger" onclick="deleteHw('${h.id}')">刪除</button>
        </div>
      `).join('');
    } else {
      document.getElementById('homeworkList').innerHTML = '暫無作業';
    }
  }
  
  function previewImage(input) {
    const file = input.files[0];
    if (!file) return;
    currentImageName = file.name;
    const reader = new FileReader();
    reader.onload = function(e) {
      currentImageBase64 = e.target.result.split(',')[1];
      document.getElementById('imagePreview').innerHTML = `<img src="${e.target.result}" class="preview-img">`;
    };
    reader.readAsDataURL(file);
  }
  
  async function publish() {
    const title = document.getElementById('title').value;
    const description = document.getElementById('description').value;
    if (!title) { alert('請輸入標題'); return; }
    
    const res = await api.publishHomework({
      title: title,
      description: description,
      imageBase64: currentImageBase64,
      imageName: currentImageName
    });
    
    if (res.success) {
      alert('作業發布成功！');
      document.getElementById('title').value = '';
      document.getElementById('description').value = '';
      document.getElementById('imagePreview').innerHTML = '';
      document.getElementById('imageFile').value = '';
      currentImageBase64 = '';
      loadHomework();
    } else {
      alert('發布失敗：' + res.message);
    }
  }
  
  async function deleteHw(id) {
    if (!confirm('確定刪除此作業？')) return;
    const res = await api.deleteHomework(id);
    if (res.success) { alert('刪除成功'); loadHomework(); }
    else alert('刪除失敗');
  }
  
  function logout() { localStorage.clear(); window.location.href = 'index.html'; }
  loadHomework();
</script>
