  <script src="js/api.js"></script>
  <script>
    let studentsData = [];
    let selectedClass = '';
    
    const urlParams = new URLSearchParams(window.location.search);
    const userType = urlParams.get('userType') || localStorage.getItem('userType') || 'teacher';
    
    async function loadClasses() {
      const response = await api.get({ action: 'getClassList' });
      if (response.success && response.classes) {
        const select = document.getElementById('classSelect');
        response.classes.forEach(cls => {
          const option = document.createElement('option');
          option.value = cls;
          option.textContent = cls;
          select.appendChild(option);
        });
      }
    }
    
    async function loadStudents() {
      selectedClass = document.getElementById('classSelect').value;
      if (!selectedClass) {
        showMessage('請選擇班級', 'error');
        return;
      }
      
      const response = await api.getStudents(selectedClass);
      if (response.success && response.students) {
        studentsData = response.students.map(s => ({ ...s, status: '未到班', note: '' }));
        renderStudentList();
        document.getElementById('statsPanel').style.display = 'grid';
        document.getElementById('tableContainer').style.display = 'block';
        document.getElementById('submitSection').style.display = 'block';
        updateStats();
      } else {
        showMessage('載入學生失敗', 'error');
      }
    }
    
    function renderStudentList() {
      const tbody = document.getElementById('studentList');
      tbody.innerHTML = '';
      studentsData.forEach((student, idx) => {
        const row = tbody.insertRow();
        row.insertCell(0).innerHTML = student.id;
        row.insertCell(1).innerHTML = student.name;
        row.insertCell(2).innerHTML = `<select class="status-select" data-idx="${idx}" onchange="updateStatus(${idx}, this.value)">
          <option value="未到班">未到班</option>
          <option value="已到班">已到班</option>
          <option value="請假">請假</option>
        </select>`;
        row.insertCell(3).innerHTML = `<input type="text" placeholder="備註" data-idx="${idx}" onchange="updateNote(${idx}, this.value)">`;
      });
    }
    
    function updateStatus(idx, status) {
      studentsData[idx].status = status;
      updateStats();
    }
    
    function updateNote(idx, note) {
      studentsData[idx].note = note;
    }
    
    function updateStats() {
      const total = studentsData.length;
      const present = studentsData.filter(s => s.status === '已到班').length;
      const absent = studentsData.filter(s => s.status === '未到班').length;
      const leave = studentsData.filter(s => s.status === '請假').length;
      document.getElementById('totalCount').innerText = total;
      document.getElementById('presentCount').innerText = present;
      document.getElementById('absentCount').innerText = absent;
      document.getElementById('leaveCount').innerText = leave;
    }
    
    async function submitAttendance() {
      const date = document.getElementById('dateSelect').value || new Date().toISOString().split('T')[0];
      const records = studentsData.map(s => ({
        date: date,
        studentId: s.id,
        studentName: s.name,
        className: selectedClass,
        status: s.status,
        note: s.note
      }));
      
      const response = await api.saveAttendance(records);
      if (response.success) {
        showMessage('點名記錄已儲存！', 'success');
      } else {
        showMessage('儲存失敗：' + response.message, 'error');
      }
    }
    
    function showMessage(msg, type) {
      const msgDiv = document.getElementById('message');
      msgDiv.textContent = msg;
      msgDiv.className = 'message ' + type;
      setTimeout(() => msgDiv.style.display = 'none', 3000);
    }
    
    function goBack() {
      window.location.href = `dashboard.html?userType=${userType}`;
    }
    
    document.getElementById('dateSelect').value = new Date().toISOString().split('T')[0];
    loadClasses();
  </script>
