<script>
  let userType = '';
  function showLogin(type) { userType = type; document.getElementById('passwordArea').style.display = 'block'; }
  async function doLogin() {
    const pwd = document.getElementById('password').value;
    const res = await api.login(userType, pwd);
    if (res.success) {
      localStorage.setItem('userType', userType);
      window.location.href = userType === 'admin' ? 'admin.html' : 'student.html';
    } else {
      document.getElementById('message').innerText = '密碼錯誤';
    }
  }
</script>
