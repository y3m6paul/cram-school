  <script src="js/api.js"></script>
  <script>
    const urlParams = new URLSearchParams(window.location.search);
    const userType = urlParams.get('userType') || localStorage.getItem('userType') || 'teacher';
    
    async function loadHomework() {
      const response = await api.getHomework();
      const container = document.getElementById('homeworkList');
      if (response.success && response.homeworks) {
        if (response.homeworks.length === 0) {
          container.innerHTML = '<p>暫無作業</p>';
          return;
        }
        container.innerHTML = response.homeworks.map(h => `
          <div class="homework-card">
            <div class="homework-title">${h.title}</div>
            <div class="homework-meta">📅 ${h.publishDate?.split('T')[0] || ''} | 班級：${h.className} | 截止：${h.deadline?.split('T')[0] || ''}</div>
            <div>${h.description || '無描述'}</div>
          </div>
        `).join('');
      } else {
        container.innerHTML = '<p>載入失敗</p>';
      }
    }
    
    async function submitHomework() {
      const data = {
        title: document.getElementById('hwTitle').value,
        description: document.getElementById('hwDesc').value,
        className: document.getElementById('hwClass').value,
        deadline: document.getElementById('hwDeadline').value
      };
      if (!data.title || !data.className) {
        showMessage('請填寫標題和班級', 'error');
        return;
      }
      const response = await api.saveHomework(data);
      if (response.success) {
        showMessage('作業發布成功！', 'success');
        document.getElementById('hwTitle').value = '';
        document.getElementById('hwDesc').value = '';
        document.getElementById('hwDeadline').value = '';
        switchTab('list');
        loadHomework();
      } else {
        showMessage('發布失敗：' + response.message, 'error');
      }
    }
    
    async function loadClasses() {
      const response = await api.get({ action: 'getClassList' });
      if (response.success && response.classes) {
        const select = document.getElementById('hwClass');
        response.classes.forEach(c => {
          const opt = document.createElement('option');
          opt.value = c;
          opt.textContent = c;
          select.appendChild(opt);
        });
      }
    }
    
    function switchTab(tab) {
      document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
      document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
      if (tab === 'list') {
        document.querySelector('.tab:first-child').classList.add('active');
        document.getElementById('list-tab').classList.add('active');
        loadHomework();
      } else {
        document.querySelector('.tab:last-child').classList.add('active');
        document.getElementById('add-tab').classList.add('active');
      }
    }
    
    function showMessage(msg, type) {
      const msgDiv = document.getElementById('message');
      msgDiv.textContent = msg;
      msgDiv.className = 'message ' + type;
      setTimeout(() => msgDiv.style.display = 'none', 3000);
    }
    
    function goBack() {
      window.location.href = `dashboard.html?userType=${userType}`;
    }
    
    loadClasses();
    loadHomework();
  </script>
